RTL-SDR multichannel streaming 3 separate NYPD frequencies to icecast simultaneously:

devices:
({
  type = "rtlsdr";
  index = 0;
  gain = 25;
  centerfreq = 476.725;
  correction = 4;
  sample_rate = 2.56;
  mode = "multichannel";
  channels:
  (
    {
      freq = 476.7625;
	  modulation = "nfm";
      outputs: 
	  (
        {
	  type = "icecast";
	  server = "icecast.example.com";
